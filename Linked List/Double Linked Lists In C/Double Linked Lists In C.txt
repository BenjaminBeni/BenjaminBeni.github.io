len(newList *l){
    newNode *n;
    int dim = 0;
    if(l -> head == NULL){
        return -1;
    }
    else{
        n = l->head;
        while(n != NULL){
            dim ++;
            n = n -> next;
        }
        return dim;
    }

}

printBack(newList *l){
    newNode *n;
    if(l->tail != NULL){
        n = l->tail;
        while(n != NULL){
            printf("%d ", n->data);
            n = n->prev;
        }
    }
    else{
        printf("%s", "List is empty.");
    }
}

print(newList *l){
    newNode *n;

        n = l->head;

            printf("%d ", n->data);
            n = n->next;
        }
    }
    else{
        printf("%s", "List is empty.");
    }

}

push(newList *l, int d){
    newNode *n;
    n = malloc(sizeof(newNode));
    if(n != NULL){
        n->data = d;
        n->next = NULL;
        n->prev = NULL;
        if(l->head == NULL){
            l->head = n;
            l->tail = n;
        }
        else{
            l->head->prev = n;
            n->next = l->head;
            l->head = n;
        }
    }
    else{
        printf("%s", "NOT OK");
    }
}

add(newList *l, int d){
    newNode *n;
    n = malloc(sizeof(newNode));
    if(n != NULL){
        n->data = d;
        n->next = NULL;
        if(l->head == NULL){
            n->prev = NULL;
            l->head = n;
            l->tail = n;
        }
        else{
            n->prev = l->tail;
            l->tail->next = n;
            l->tail = n;
        }
    }
    else{
        printf("%s", "NOT OK");
    }
}

insert(newList *l, int d, int poz){
    newNode *n;
    if(l->head == NULL){
        push(&L, d);
    }
    else{
        if(poz == 0){
            push(&L, d);
        }
        else{
            int count = 0;
            n = l->head;
            while(n != NULL){
                count++;
                if(count == poz){
                    break;
                }
                n = n->next;
                if(n == NULL){
                    add(&L, d);
                    return;
                }
            }
            newNode *tmp;
            tmp = malloc(sizeof(newNode));
            tmp->data = d;
            tmp->prev = n;
            tmp->next = n->next;
            n->next = tmp;
        }
    }
}

del(newList *l, int poz){
    newNode *n;
    if(l->head == NULL){
        return;
    }
    else{
        if(poz == 0){
            l->head = l->head->next;
        }
        else{
            int count = 0;
            n = l->head;
            while(n != NULL){
                if(count == poz){
                    break;
                }
                n = n->next;
                if(n->next == NULL){
                    n->prev->next = NULL;
                    return;
                }
                count++;
            }
            n->prev->next = n->next;
            n->next->prev = n->prev;
        }
    }
}

retVal(newList *l, int arr[]){
    newNode *n;
    int c = 0;
    if(l->head != NULL){
        n = l->head;
        while(n != NULL){
            arr[c] = n->data;
            c++;
            if(n->next == NULL){
               arr[c] = -1;
            }
            n = n->next;
        }
    }
}

searchS(newList *l, int val){
    newNode *n;
    int count = 0;
    if(l -> head == NULL){
        printf("%s", "List is empty");
    }
    else{
        n = l -> head;
        while(n != NULL){
            if( n -> data == val ){
                return count;
            }
            else{
                count++;
            }
            n = n -> next;
        }
    }
    return -1;
}

merge2L(newList *l1, newList *l2){
    newNode *n2 = l2->head;
    l1->tail->next = n2;
    while(n2 != NULL){
        add(l1, n2->data);
        n2 = n2->next;
    }
}
